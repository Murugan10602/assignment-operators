# assignment-operators
#assignment operators:
"""
= assignment 
+= addition
-= subtraction
*= multiplication
/= division
%= modulus
**= exponentiation
//= floor division

"""
a=120
print(a)
a+=5       #a=a+5
print(a)
a-=10      #a=a-10
print(a)
a*=5       #a=a*5
print(a)
a/=5       #a=a/5
print(a)
a%=5       #a=a%5
print(a)
a//=5      #a=a//5
print(a)
a**=5      #a=a**5
print(a)
